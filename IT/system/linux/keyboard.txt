##remap Super_L##
#find Super_L keycode with
$ xmodmap -pk
#(probably 133)

$ xmodmap -e "remove mod4 = Super_L"
#find free keycode with
$ xev
#in my case 148
$ xmodmap -e "keycode 133 = 148"
