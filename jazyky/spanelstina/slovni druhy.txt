slovesa {

  koncovka -AR ex. [hablar - mluvit, trabajar -	pracovat, pensar - přemýšlet]
  (o, as, a, amos, áis, an) => trabajar[trabajo, trabajas, trabaja, trabajamos, trabajáis, trabajan]

  koncovka -ER ex. comer(jíst) correr(běžet) beber(pít)
  (o, es, e, emos, éis, en) => comer[como, comes, come, comemos, coméis, comen]

  koncovka -IR ex. vivir(žít) escribir(psát) abrir(otevřít) conducir(řídit)
  (o, es, e, imos, ís, en) => vivir[vivo, vives, vive, vivimos, vivís, viven]

  minulý čas - "haber" (pomocné sloveso):
    (he, has, ha, hemos, habéis, han) {
      Yo he hablado
      Tú has hablado
      Él/Ella/Usted ha hablado
      Nosotros/Nosotras hemos hablado
      Vosotros/Vosotras habéis hablado
      Ellos/Ellas/Ustedes han hablado
    }

    -AR hablar → hablado (mluvit → mluvěno) ar -> ado
    -ER comer → comido (jíst → jedeno) er -> ido
    -IR vivir → vivido (žít → žito) ir -> ido


  Být - Ser (soy, eres, es, somos, sois, son) -  (jsem, jsi, je, jsme, jste, jsou) #(trvalé stavy, identita, čas, původ)
